67 => "0110000000000000",	-- OR R0, R0, R0
68 => "0110000000000000",	-- OR R0, R0, R0
69 => "0110000000000000",	-- OR R0, R0, R0
70 => "0110000000000000",	-- OR R0, R0, R0
